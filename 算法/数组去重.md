# 数组去重
## 数组里的数组去重
```js
var arr=[[1,2],[2,1]]
function qu(carr) {
    for (let i = 0; i < carr.length - 1; i++) {
        var like = false;
        for (let j = i + 1; j < carr.length; j++) {
            if (carr[i].length == carr[j].length) {
                // 两两比较。若长度遍历内部比较
                var n = 0;
                for (let k = 0; k < carr[i].length; k++) {
                    for (let l = 0; l < carr[j].length; l++) {
                        if (carr[i][k] == carr[j][l]) {
                            n++;
                        }
                    }
                    // 相似数量与数组长度一致，则这个元素重复了。
                    if (n == carr[j].length) {
                        carr.splice(j, 1);
                        j--;
                    }
                }
            }
        }
    }
    return carr
}
```
